import math
import numpy as np
import matplotlib.pyplot as plt
from mpl_toolkits.mplot3d import Axes3D
from scipy.optimize import minimize

# Define a proxy GCF function for resonance optimization
def gcf_resonance(config, room_dims, target_freq=1.0):
    """
    Proxy GCF to compute 'resonance energy' for ball configurations.
    Lower energy = better constructive harmony (minimize dissonance).
    Config: flattened array of ball positions (x,y,z) and radii.
    Room_dims: [length, width, height]
    Target_freq: Proxy for 'desired resonance' (e.g., health=1.0, peace=0.5)
    """
    num_balls = len(config) // 4  # x,y,z,r per ball
    positions = config[:3*num_balls].reshape(num_balls, 3)
    radii = config[3*num_balls:]

    # SS proxy: pairwise distances/inverse for 'interaction'
    dists = np.linalg.norm(positions[:, np.newaxis] - positions, axis=-1) + np.diag([np.inf]*num_balls)
    ss = np.sum(1 / dists)  # Unified SS (absolute/net integrated)

    # SSG proxy: gradient as position variances
    ssg = np.var(positions, axis=0).sum()

    # Entropy proxy: maximize 'diversity' (radii variance)
    entropy = -np.var(radii)  # Negative for minimization

    # EA proxy: total 'energy' from radii sums
    ea = np.sum(radii) - target_freq * num_balls  # Bias to target

    # GR proxy: penalize non-resonant ratios (golden ratio phi)
    phi = (1 + math.sqrt(5)) / 2
    ratio_penalties = np.sum(np.abs(radii[:, np.newaxis] / radii - phi))

    # EMTT proxy: threshold tipping (if below, high penalty)
    tipping_threshold = 0.1 * num_balls
    if ss < tipping_threshold:
        return np.inf  # No tip

    # Objective: minimize 'dissonance' (negative resonance)
    dissonance = ss + ssg - entropy + abs(ea) + ratio_penalties
    return dissonance

# Optimization to find ball config
def optimize_config(num_balls, room_dims, target_freq=1.0):
    """
    Use scipy minimize to find optimal positions/radii.
    Bounds: positions in room, radii 0.1-1.0 m.
    """
    # Initial guess: random in room
    init_pos = np.random.uniform(0, room_dims, (num_balls, 3)).flatten()
    init_radii = np.random.uniform(0.1, 1.0, num_balls)
    init_config = np.concatenate([init_pos, init_radii])

    # Bounds
    pos_bounds = [(0, dim) for dim in room_dims] * num_balls
    rad_bounds = [(0.1, 1.0)] * num_balls
    bounds = pos_bounds + rad_bounds

    # Minimize
    res = minimize(gcf_resonance, init_config, args=(room_dims, target_freq),
                   bounds=bounds, method='L-BFGS-B')

    return res.x, res.fun

# Visualize
def visualize_balls(config, room_dims):

import matplotlib.pyplot as plt
from scipy.signal import hilbert

def simulate_eeg_entrainment(num_steps=1000, freq_target=10.0, ss_bias=0.1, tipping_threshold=0.8):
    """
    Simulate EEG wave entrainment between two partners.
    Waves as sinusoidal with phase noise; SS proxy biases alignment.
    EM tipping: If coherence > threshold, "consonance" achieved.
    Output: Waves, coherence over time.
    """
    t = np.linspace(0, num_steps / freq_target, num_steps)

    # Initial waves with dissonance (phase offset, noise)
    phase_offset = np.pi / 2  # Initial dissonance
    wave1 = np.sin(2 * np.pi * freq_target * t) + np.random.normal(0, 0.1, num_steps)

    # Simulate entrainment: Bias phases toward sync

        bias = ss_bias * np.sin(phase_diff)  # SSG-like pull

        wave2[step] += bias * (wave1[step] - wave2[step])

            print(f"Step {step}: Consonance tipped (coherence {corr:.2f})")

    plt.plot(t, wave2, label='Partner 2')
    plt.xlabel('Time (s)')
    plt.ylabel('EEG Amplitude')
    plt.title('EEG Wave Entrainment Simulation')
    plt.legend()
    plt.show()

    return coherence
